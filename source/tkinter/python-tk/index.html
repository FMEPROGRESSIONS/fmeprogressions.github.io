from tkinter import * # Импортируем TKinter
from tkinter import messagebox as mb


def none ():
    print("Кнопка нажата!")

def Info ():
    Text = Input.get () # Забираем текст из нашего поля инпут внизу

    Text_str = str (Text) # конвертируем этот текст в строках

    mb.showinfo(title="Заголовок", message=Text_str) # вывод всплывающего окна

def Err ():

    Text = Input.get () # Забираем текст из нашего поля инпут внизу

    Text_str = str (Text) # конвертируем этот текст в строках

    mb.showerror(title="Заголовок", message=Text_str) # вывод всплывающего окна

def Warn ():

    Text = Input.get () # Забираем текст из нашего поля инпут внизу

    Text_str = str (Text) # конвертируем этот текст в строках

    mb.showwarning(title="Заголовок", message=Text_str) # вывод всплывающего окна

root = Tk()

root['bg'] = '#3b3737' # Устанавливаем цвет окна root.title('Name') # Устанавливаем заголовок окна
root.wm_attributes('-alpha', 1) # Устанавливаем прозрачность
root.geometry('300x250') # Устанавливаем размер в px
root.resizable(width=False, height=False) # Здесь мы говорим, что размер окна менять нельзя

canvas = Canvas(width=300, height=250, bg='#3b3737')
canvas.pack()

frame = Frame (root, bg = '#3b3737')
frame.place (relwidth = 1, relheight = 1)

mainmenu = Menu(root) # Инициализируем меню
filemenu = Menu(mainmenu, tearoff=0) # Убираем странную полоску сверху, открепляющую меню
root.config(menu=mainmenu)
mainmenu.add_command(label="Информация", command=Info) # Пункт
mainmenu.add_command(label="Ошибка", command=Err) # Пункт
mainmenu.add_command(label="Предупреждение", command=Warn) # Все еще пункт


title = Label(frame, text='Это текст', font='Comfortaa 20', fg = '#FFFFFF', bg='#3b3737', pady=5) # Добавляем текст
title.pack() # Отображаем текст

Input = Entry (frame, bg = '#302D2D') # Поле ввода
Input.pack () # Отображение поля ввода

root.mainloop() # Запускаем основной цикл, чтобы программа работала
