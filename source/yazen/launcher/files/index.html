 # Подключаем нужные библиотеки
from tkinter import * # Импортируем TKinter
import tkinter.messagebox as mb
### ПЕРЕМЕННЫЕ
root = Tk() # Создаем переменную типа Tk
WIDTH = 300
HEIGHT = 250
WD = '300x250'
# colors
RED = '#FF0000'
LIGHT_RED = '#FF4500'
ORANGE = '#FFA500'
YELLOW = '#FFFF00'
GREEN = '#00FF00'
BLUE = '#0000FF'
LIGHT_BLUE = '#ADD8E6'
PURPLE = '#800080'
BLACK = '#000000'
WHITE = '#FFFFFF'
LIGHT_GRAY = '#D3D3D3'

### Функции для кнопок
def but_inf(): # Функция при нажатии на кнопку
text = Input.get() # Забираем текст из нашего поля инпут внизу
info_str = str(text) # конвертируем этот текст в строк
# ИНФО
mb.showinfo(title="Название всплывающего окна", message=info_str) # выводим текст в всплывающем окне
def but_war(): # Функция при нажатии на кнопку
text = Input.get()
info_str = str(text)
# ПРЕДУПРЕЖДЕНИЕ
mb.showwarning(title="Название всплывающего окна", message=info_str)
def but_err(): # Функция при нажатии на кнопку
text = Input.get()
info_str = str(text)
# ОШИБКА
mb.showerror(title="Название всплывающего окна", message=info_str)
###

### ОКНО
root['bg'] = LIGHT_GRAY # Устанавливаем цвет окна
root.title('none') # Устанавливаем заголовок окна
root.wm_attributes('-alpha', 1) # Устанавливаем прозрачность

root.geometry(WD) # Устанавливаем размер в px
root.resizable(width=False, height=False) # Здесь мы говорим, что размер окна менять нельзя

canvas = Canvas(root, height=HEIGHT, width=WIDTH) # Устанавливаем канвас, это что-то
canvas.pack() # вроде холста, на который мы повесим виджеты

frame = Frame(root, bg='gray') # Фон того самого холста
frame.place(relwidth=1, relheight=1) # Смещение и размер холста
###

### СОДЕРЖИМОЕ ОКНА
# ТЕКСТ
title = Label(frame, text='Введите текст и выберите тип окна: ', bg='gray', font=140) # Установим текст
title.pack() # Отображение текста
# ПОЛЕ ВВОДА
Input = Entry(frame, bg='white') # Поле ввода
Input.pack() # Отображение поля ввода
# КНОПКА
but_inf = Button(frame, text='Информация', bg='green', command=but_inf) # Установим кнопку и действие-функцию
but_inf.pack() # Отображение кнопки
but_war = Button(frame, text='Предупреждение', bg='green', command=but_war) # Установим кнопку и действие-функцию
but_war.pack() # Отображение кнопки
but_err = Button(frame, text='Ошибка', bg='green', command=but_err) # Установим кнопку и действие-функцию
but_err.pack() # Отображение кнопки
###
root.mainloop() # Запускаем основной цикл, чтобы программа работала
